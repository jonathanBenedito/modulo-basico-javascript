// Função com parâmetros

function soma(numero1, numero2) {
    return numero1 + numero2
}

alert(soma(1, 3))

// Função de parâmetros com valor padrão

function nome(nomeDoUsuario = 'Anônimo') {
    return nomeDoUsuario
}

alert(nome("Roberto"))
