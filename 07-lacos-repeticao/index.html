Laços de Repetição

// for(inicializador; condição; incremento)

for(let i = 0; i < 5; i++) {
  console.log(i)
}

// while(condição)

let a = 10 let b = 15

while(a < b) {
  console.log('incrementando a variavel a ' + a)
  a++
}

// do {} while(condição)

var i = 1

do{
  console.log('entrou ' + i)
  i++
}while(i <= 5)
